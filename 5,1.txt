int check(vector<int> a, int d)
{
    int k = -1;
    for (int i = 0; i < a.size(); i++)
    {
        if (a[i] == d)
            k = i;
    }
    if (k == -1)
        k = a.size() - 1;
    return k;
}
void bum(vector<int> a ,int* num, int* min,int n)
{
    *min = a[0];
    *num = 1;
    for (int i = 0; i < n; i++)
    {
        if (a[i] < *min)
        {
            *min = a[i];
            *num = i + 1;
        }
    }
}
